class BinaryRelevanceClassifier(BaseEstimator, ClassifierMixin):
    """The BinaryRelevanceClassifier is an algorithm that uses 
    a homogenous ensemble of models to predict multi-output labels. 
    It uses a One-vs-All approach to train the ensemble of models.

    Parameters
    ----------
    classifier: (sklearn classifier), optional (default = DecisionTreeClassifier(criterion='entropy'))
    What classifier to use internally
    
    """
    def __init__(self, classifier=None):
        # Check if it is a valid classifier
        if classifier is None:
            self.classifier = DecisionTreeClassifier(criterion='entropy')
        elif is_classifier(classifier):
            self.classifier = classifier
        else:
            raise ValueError("'classifier' needs to be an sklearn Classifier")

    def fit(self, X, y):
        # Check that X and y have correct shape
        X, y = check_X_y(X, y, multi_output=True)
        # Ensure that target y is of a non-regression type
        check_classification_targets(y)
        
        self.number_of_output_classes = y.shape[1]
        
        # Train the ensemble of models
        self.models = []
        for i in range(self.number_of_output_classes):
            # Separate the target data
            target = y[:,i]
            # Initialise and fit the model
            m = estimator_clone(self.classifier)
            m.fit(X, target)
            self.models.append(m)
        
        return self

    def predict(self, X):
        # Check if fit had been called
        check_is_fitted(self, ['number_of_output_classes', 'models'])

        # Check that the input features match the type and shape of the training features
        X = check_array(X)

        # Predict using the ensemble of trained models
        predictions = []
        for i in range(self.number_of_output_classes):
            y_pred = self.models[i].predict(X)
            predictions.append(y_pred)
        
        # Fixing the shape of predictions array
        predictions = np.array(predictions).T
        
        return predictions

    def predict_proba(self, X):
        # Check if fit had been called
        check_is_fitted(self, ['number_of_output_classes', 'models'])

        # Check that the input features match the type and shape of the training features
        X = check_array(X)

        # Generate prediction probabilities using the ensemble of trained models
        pred_probas = []
        for i in range(self.number_of_output_classes):
            y_pred = self.models[i].predict_proba(X)
            pred_probas.append(y_pred)
        
        # Fixing the shape of pred_probas array
        pred_probas = np.stack(pred_probas, axis=1)
        
        return pred_probas

class BinaryRelevanceClassifier(BaseEstimator, ClassifierMixin):
    """The BinaryRelevanceClassifier is an algorithm that uses 
    a homogenous ensemble of models to predict multi-output labels. 
    It uses a One-vs-All approach to train the ensemble of models.

    Parameters
    ----------
    classifier: (sklearn classifier), optional (default = DecisionTreeClassifier(criterion='entropy'))
    What classifier to use internally
    
    undersampler: (imblearn undersampler), optional (default = None)
    What imblearn based undersampling technique to use
    
    """
    def __init__(self, classifier=None, undersampler=None):
        
        # Check if it is a valid classifier
        if classifier is None:
            self.classifier = DecisionTreeClassifier(criterion='entropy')
        elif is_classifier(classifier):
            self.classifier = classifier
        else:
            raise ValueError("'classifier' needs to be an sklearn Classifier")

        # Check if it is a valid imblearn undersampler
        if undersampler is None:
            self.undersampler = None
        elif hasattr(undersampler, '_sampling_type') and undersampler._sampling_type in ['under-sampling', 'clean-sampling']:
            self.undersampler = undersampler
        else:
            raise ValueError("'undersampler' needs to be an imblearn undersampler")
    
    def fit(self, X, y):
        # Check that X and y have correct shape
        X, y = check_X_y(X, y, multi_output=True)
        # Ensure that target y is of a non-regression type
        check_classification_targets(y)
        
        self.number_of_output_classes = y.shape[1]
        
        # Train the ensemble of models
        self.models = []
        for i in range(self.number_of_output_classes):
            # Separate the target data
            target = y[:,i]
            
            # Initialise and fit the model
            m = estimator_clone(self.classifier)
            
            if self.undersampler is not None:
                # Undersampling the data according to the target labels
                sampled_X, sampled_y = self.undersampler.fit_resample(X, target)
                m.fit(sampled_X, sampled_y)
            else:
                m.fit(X, target)
            
            self.models.append(m)
        
        return self

    def predict(self, X):
        # Check if fit had been called
        check_is_fitted(self, ['number_of_output_classes', 'models'])

        # Check that the input features match the type and shape of the training features
        X = check_array(X)

        # Predict using the ensemble of trained models
        predictions = []
        for i in range(self.number_of_output_classes):
            y_pred = self.models[i].predict(X)
            predictions.append(y_pred)
        
        # Fixing the shape of predictions array
        predictions = np.array(predictions).T
        
        return predictions

    def predict_proba(self, X):
        # Check if fit had been called
        check_is_fitted(self, ['number_of_output_classes', 'models'])

        # Check that the input features match the type and shape of the training features
        X = check_array(X)

        # Generate prediction probabilities using the ensemble of trained models
        pred_probas = []
        for i in range(self.number_of_output_classes):
            y_pred = self.models[i].predict_proba(X)
            pred_probas.append(y_pred)
        
        # Fixing the shape of pred_probas array
        pred_probas = np.stack(pred_probas, axis=1)
        
        return pred_probas

class ClassifierChains(BaseEstimator, ClassifierMixin):
    """The ClassifierChains is an algorithm that uses a 
    homogenous ensemble of models to predict multi-output labels. 
    It uses a One-vs-All approach to train the first model. Then,
    the output of this model is provided as an input to the next 
    model. This process is repeated for all subsequent models. 

    Parameters
    ----------
    classifier: (sklearn classifier), optional (default = DecisionTreeClassifier(criterion='entropy'))
    What classifier to use internally
    
    undersampler: (imblearn undersampler), optional (default = None)
    What imblearn based undersampling technique to use
    
    """
    def __init__(self, classifier=None, undersampler=None):
        
        # Check if it is a valid classifier
        if classifier is None:
            self.classifier = DecisionTreeClassifier(criterion='entropy')
        elif is_classifier(classifier):
            self.classifier = classifier
        else:
            raise ValueError("'classifier' needs to be an sklearn Classifier")

        # Check if it is a valid imblearn undersampler
        if undersampler is None:
            self.undersampler = None
        elif hasattr(undersampler, '_sampling_type') and undersampler._sampling_type in ['under-sampling', 'clean-sampling']:
            self.undersampler = undersampler
        else:
            raise ValueError("'undersampler' needs to be an imblearn undersampler")

    def fit(self, X, y):
        # Check that X and y have correct shape
        X, y = check_X_y(X, y, multi_output=True)
        # Ensure that target y is of a non-regression type
        check_classification_targets(y)
        
        self.number_of_output_classes = y.shape[1]
        
        # Randomly ordering the target labels
        self.y_order = list(range(self.number_of_output_classes))
        random.shuffle(self.y_order)
        
        # Train the ensemble of models
        self.models = [None for _ in range(self.number_of_output_classes)]
        
        data = X
        for index in self.y_order:
            # Separate the target data
            target = y[:,index]
            
            # Initialise and fit the model
            m = estimator_clone(self.classifier)
            
            if self.undersampler is not None:
                # Undersampling the data according to the target labels
                sampled_X, sampled_y = self.undersampler.fit_resample(data, target)
                m.fit(sampled_X, sampled_y)
            else:              
                m.fit(data, target)
                
            self.models[index] = m
            
            # Append the target column of this iteration to the data set
            data = np.hstack([data, np.array([target]).T])
        
        return self

    def predict(self, X):
        # Check if fit had been called
        check_is_fitted(self, ['number_of_output_classes', 'models', 'y_order'])

        # Check that the input features match the type and shape of the training features
        X = check_array(X)

        # Predict using the ensemble of trained models
        predictions = [None for _ in range(self.number_of_output_classes)]
        
        data = X
        for index in self.y_order:
            y_pred = self.models[index].predict(data)
            predictions[index] = y_pred
            
            # Append the predicted column of this iteration to the data set
            data = np.hstack([data, np.array([y_pred]).T])
        
        return np.array(predictions).T

    def predict_proba(self, X):
        # Check if fit had been called
        check_is_fitted(self, ['number_of_output_classes', 'models', 'y_order'])

        # Check that the input features match the type and shape of the training features
        X = check_array(X)

        # Generate prediction probabilities using the ensemble of trained models
        pred_probas = [None for _ in range(self.number_of_output_classes)]
        
        data = X
        for index in self.y_order:
            pred_prob = self.models[index].predict_proba(data)
            pred_probas[index] = pred_prob
            
            # Generate the prediction
            y_pred = self.models[index].predict(data)
            
            # Append the predicted column of this iteration to the data set
            data = np.hstack([data, np.array([y_pred]).T])

        # Fixing the shape of pred_probas array
        pred_probas = np.stack(pred_probas, axis=1)
        
        return pred_probas

Task 6: Reflect on the Performance of the Different Models Evaluated¶

The evaluation measure should be dependent on the problem being solved. To evaluate performance in multi label classification, we can use measures like exact matching, hamming loss or macro-averaged F1 score. Exact matching considers the entire set of predicted labels for a given sample to be incorrect if it does not entirely match the true set. Hamming loss is a slightly loose measure as it penalises only the individual labels. However for imbalanced problems, accuracy is a problematic metric and in order to be able to account for the class imbalance, we used the macro-averaged F1 score as our primary evaluation measure.

For our evaluation experiment, we have split the data into train and test sets. We have used the training set to perform a Grid Search with cross validation to choose the best base classifier and undersampling technique. In order to utilise maximum amount of data for training, we decided to use cross validation. Once chosen, we retrained the model with best parameters and evaluated the model on the test set.

Over several runs of the Grid Search on BinaryRelevanceAlgorithm and ClassifierChains, we have noticed that RandomForestClassifier and SupportVectorClassifier perform best along with RandomUndersampling technique.

The BinaryRelevance and ClassifierChain algorithms have the complexity of the base model chosen multiplied by the number of labels to be classified.

Since classifier chains are highly susceptible to the order of labels chosen, in order to figure out an optimum label order, we tried:

  • Random Shuffling
  • Based on pairwise correlation using Pearson, Kendall Tau and Spearman rank correlation coefficients.
  • Based on individual model accuracies in Binary Relevance classifier.
  • A self implemented algorithm based on Quick Sorting Algorithm

We observed that Random Shuffling worked best and thus we included it in the ClassifierChain class implementation. We have not included other techniques in the notebook.
